# [12. 八皇后](https://leetcode-cn.com/problems/eight-queens-lcci/)

## 题目

![BvJRMu](https://gitee.com/yoyhm/oss/raw/master/uPic/BvJRMu.png)

## 分析

本题考查 **回溯** 找到所有的可能。

## 代码

```C++
class Solution {
public:
    vector<vector<string>> ans;
    vector<vector<string>> solveNQueens(int n) {
        vector<string> path(n, string(n, '.'));
        backtrack(n, 0, path);
        return ans;
    }

    bool isLegal(int row, int col, int n, vector<string>& path) {
        // 判断列是否有一样的
        for (int i = 0; i < row; i++) {
            if (path[i][col] == 'Q') return false;
        }
        // 判断右上角
        for (int i = row - 1, j = col - 1; i >= 0 && j >= 0; i--, j--) {
            if (path[i][j] == 'Q') return false;
        }
        // 判断左上角
        for (int i = row - 1, j = col + 1; i >= 0 && j < n; i--, j++) {
            if (path[i][j] == 'Q') return false;
        }
        return true;
    }
    void backtrack(int n, int row, vector<string>& path) {
        if (n == row) {
            ans.push_back(path);
            return;
        }
        for (int col = 0; col < n; col++) {
            if (isLegal(row, col, n, path)) {
                path[row][col] = 'Q';
                backtrack(n, row + 1, path);
                path[row][col] = '.';
            }
        }
    }

};
```
